def matrixmul(A, B):
    """æ‰‹å†™çŸ©é˜µä¹˜æ³• A x B"""
    m, n, p, q = len(A), len(A[0]), len(B[0]), len(B)
    if m != p or n != q:
        return -1

    C = [[0] * p for _ in range(m)]
    for i in range(m):
        for j in range(p):
            for k in range(n):
                C[i][j] += A[i][k] * B[k][j]
    return C
